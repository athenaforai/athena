35 void athena::backend::generic::GenericExecutor::processBytecode (
36  athena::backend::VMState &state, std::vector<vm_word> bytecode ) {
37  OpCode op;
38  auto *args = new vm_word[10];
39  unsigned long argc = 0;
40 
41  auto gmm = dynamic_cast<GenericMemoryManager*>(device->getMemoryManager());
42 
43  while ( parse( bytecode, state.IP, op, args, argc ) <= bytecode.size()) {
44  switch ( op ) {
45  case OpCode::ADD: {
46  gmm->loadAndLock( args[ 0 ] );
47  gmm->loadAndLock( args[ 1 ] );
48  gmm->allocateAndLock( args[ 2 ] );
49  add( gmm,
50  gmm->getTensor( args[ 0 ] ),
51  gmm->getTensor( args[ 1 ] ),
52  gmm->getTensor( args[ 2 ] ));
53 
54  gmm->unlock( args[ 0 ] );
55  gmm->unlock( args[ 1 ] );
56  gmm->unlock( args[ 2 ] );
57  break;
58  }
59  case OpCode::MATMUL: {
60 
61  gmm->loadAndLock( args[ 1 ] );
62  gmm->loadAndLock( args[ 3 ] );
63  gmm->allocateAndLock( args[ 4 ] );
64 
65  auto aTransp = static_cast<bool>(args[ 0 ]);
66  Tensor* a = gmm->getTensor( args[ 1 ] );
67  auto bTransp = static_cast<bool>( args[ 2 ]);
68  Tensor* b = gmm->getTensor( args[ 3 ] );
69  Tensor* res = gmm->getTensor( args[ 4 ] );
70  matmul( gmm, aTransp, a, bTransp, b, res );
71 
72  gmm->unlock( args[ 1 ] );
73  gmm->unlock( args[ 3 ] );
74  gmm->unlock( args[ 4 ] );
75  break;
76  }
77  case OpCode::SIGMOID: {
78  gmm->loadAndLock( args[ 0 ] );
79  gmm->allocateAndLock( args[ 1 ] );
80  Tensor* x = gmm->getTensor( args[ 0 ]);
81  Tensor* res = gmm->getTensor( args[ 1 ]);
82  sigmoid( gmm, x, res );
83  gmm->unlock( args[ 0 ] );
84  gmm->unlock( args[ 1 ] );
85  break;
86  }
87  case OpCode::SIGMOID_DERIV: {
88  gmm->loadAndLock( args[ 0 ] );
89  gmm->allocateAndLock( args[ 1 ] );
90  Tensor* x = gmm->getTensor( args[ 0 ]);
91  Tensor* res = gmm->getTensor( args[ 1 ]);
92  sigmoid_deriv( gmm, x, res );
93  gmm->unlock( args[ 0 ] );
94  gmm->unlock( args[ 1 ] );
95  break;
96  }
97  case OpCode::SCALE: {
98  gmm->loadAndLock( args[ 0 ] );
99  gmm->loadAndLock( args[ 1 ] );
100  gmm->allocateAndLock( args[ 2 ] );
101  Tensor* x = gmm->getTensor( args[ 0 ]);
102  Tensor* y = gmm->getTensor( args[ 1 ]);
103  Tensor* res = gmm->getTensor( args[ 2 ]);
104  scale( gmm, x, y, res );
105  gmm->unlock( args[ 0 ] );
106  gmm->unlock( args[ 1 ] );
107  gmm->unlock( args[ 2 ] );
108  break;
109  }
110  case OpCode::MSE: {
111  gmm->loadAndLock( args[ 0 ] );
112  gmm->loadAndLock( args[ 1 ] );
113  gmm->allocateAndLock( args[ 2 ] );
114  Tensor* x = gmm->getTensor( args[ 0 ]);
115  Tensor* y = gmm->getTensor( args[ 1 ]);
116  Tensor* res = gmm->getTensor( args[ 2 ]);
117  mse( gmm, x, y, res );
118  gmm->unlock( args[ 0 ] );
119  gmm->unlock( args[ 1 ] );
120  gmm->unlock( args[ 2 ] );
121  break;
122  }
123  case OpCode::MSE_DERIV: {
124  gmm->loadAndLock( args[ 0 ] );
125  gmm->loadAndLock( args[ 1 ] );
126  gmm->allocateAndLock( args[ 2 ] );
127  Tensor* x = gmm->getTensor( args[ 0 ]);
128  Tensor* y = gmm->getTensor( args[ 1 ]);
129  Tensor* res = gmm->getTensor( args[ 2 ]);
130  mse_deriv( gmm, x, y, res );
131  gmm->unlock( args[ 0 ] );
132  gmm->unlock( args[ 1 ] );
133  gmm->unlock( args[ 2 ] );
134  break;
135  }
136  case OpCode::MKSCALAR: {
137  gmm->allocateAndLock( args[ 1 ] );
138  unsigned long x = args[ 0 ];
139  Tensor* res = gmm->getTensor( args[ 1 ]);
140  mkscalar( gmm, x, res );
141  gmm->unlock( args[ 1 ] );
142  break;
143  }
144  case OpCode::TRANSPOSE: {
145  gmm->loadAndLock( args[ 0 ] );
146  gmm->allocateAndLock( args[ 1 ] );
147  Tensor* x = gmm->getTensor( args[ 0 ]);
148  Tensor* res = gmm->getTensor( args[ 1 ]);
149  transpose( gmm, x, res );
150  gmm->unlock( args[ 0 ] );
151  gmm->unlock( args[ 1 ] );
152  break;
153  }
154  case OpCode::COPY: {
155  gmm->loadAndLock( args[ 0 ] );
156  gmm->allocateAndLock( args[ 1 ] );
157  Tensor* x = gmm->getTensor( args[ 0 ]);
158  Tensor* res = gmm->getTensor( args[ 1 ]);
159  copy( gmm, x, res );
160  gmm->unlock( args[ 0 ] );
161  gmm->unlock( args[ 1 ] );
162  break;
163  }
164  case OpCode::DEL: {
165  gmm->deleteFromMem( args[ 0 ] );
166  break;
167  }
168  case OpCode::ALLOC: {
169  auto shape = new TensorShape( args, argc - 1 );
170  auto t = new Tensor( *shape, DataType::FLOAT ); // todo data type
171  t->setStartAddress( args[ argc - 1 ] );
172  gmm->addTensor( t );
173  break;
174  }
175  case OpCode::MUL: {
176  gmm->loadAndLock( args[ 0 ] );
177  gmm->loadAndLock( args[ 1 ] );
178  gmm->allocateAndLock( args[ 2 ] );
179  mul( gmm,
180  gmm->getTensor( args[ 0 ] ),
181  gmm->getTensor( args[ 1 ] ),
182  gmm->getTensor( args[ 2 ] ));
183 
184  gmm->unlock( args[ 0 ] );
185  gmm->unlock( args[ 1 ] );
186  gmm->unlock( args[ 2 ] );
187  break;
188  }
189  case OpCode::HADAMARD: {
190  gmm->loadAndLock( args[ 0 ] );
191  gmm->loadAndLock( args[ 1 ] );
192  gmm->allocateAndLock( args[ 2 ] );
193  hadamard( gmm,
194  gmm->getTensor( args[ 0 ] ),
195  gmm->getTensor( args[ 1 ] ),
196  gmm->getTensor( args[ 2 ] ));
197 
198  gmm->unlock( args[ 0 ] );
199  gmm->unlock( args[ 1 ] );
200  gmm->unlock( args[ 2 ] );
201  break;
202  }
203  default:
204  throw std::runtime_error( "Unknown instruction" );
205  }
206  }
