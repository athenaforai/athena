43  /*
44  * The whole algorithm:
45  * 1. Calculate actual error E
46  * 2. Let's Q - queue of nodes, EQ - queue of errors
47  * 2.1 node -> Q
48  * 2.2 E -> EQ
49  * 3. For each node N in Q
50  * 3.1 E = EQ.top()
51  * 3.2 If this is variable node, adjust weights: N = N - alpha * E
52  * 3.3 If this is regular node, for each incoming node I:
53  * 3.3.1 If I is constant, skip
54  * 3.3.2 Ei = Di (*) E, where Ei is the new error value, Di - derivative of N
55  * with respect to I, (*) - Hadamard (elementwise) product.
56  * 3.3.3 Ei -> EQ
57  * 3.3.4 I -> Q
58  */

62  std::vector< vm_word > tmpV( 1 );
