33 athena::core::Session::getByteCode ( Node* logits ) {
34 
35  std::vector< vm_word > curBC;
36  vm_word resultCell = 0;
37  TensorShape * resultShape;
38 
39  if ( logits->isInputNode()) {
40  auto inpNode = dynamic_cast<InputNode*>(logits);
41  resultShape = new TensorShape(inpNode->getData()->getShape());
42  resultCell = virtualMemory->allocate( inpNode->getData());
43  inpNode->setMappedMemCell( resultCell );
44  headNodes.push_back( inpNode );
45  } else {
46  if ( logits->getOp()->getOperandsCount() !=
47  logits->getIncomingNodes().size()) {
48  throw std::runtime_error(
49  "Wrong count of arguments at " + logits->getName() + ". Expected " +
50  std::to_string( logits->getOp()->getOperandsCount()) + ", got " +
51  std::to_string( logits->getIncomingNodes().size()));
52  } else {
53 
54  // todo check shape compatibility
55  std::vector< vm_word > resCells;
56  std::vector< TensorShape > predShapes;
57  for (
58  Node* pred : logits->getIncomingNodes()) {
59  if ( pred->isCalculated()) {
60  resCells.push_back( pred->getResult());
61  pred->updateUsageCount();
62  if ( pred->isGarbage()) {
63  virtualMemory->free( pred->getResult());
64  curBC.push_back( static_cast<vm_word>( OpCode::DEL ));
65  curBC.push_back( pred->getResult());
66  }
67  } else {
68  auto[predBC, predShape, predResCell] = getByteCode( pred );
69  curBC.insert(
70  std::end( curBC ), std::begin( predBC ),
71  std::end( predBC ));
72  resCells.push_back( predResCell );
73  predShapes.push_back( *predShape );
74  }
75  }
76 
77  resultShape = new TensorShape(logits->getOp()->getOutputShape( predShapes ));
78  auto resultTensor = new Tensor( *resultShape, DataType::FLOAT ); // todo TYPE
79  resultCell = virtualMemory->allocate( resultTensor );
80 
81  curBC.push_back( static_cast<vm_word>( OpCode::ALLOC ));
82  curBC.push_back( resultShape->dimensions());
83  for ( unsigned int d = 0; d < resultShape->dimensions(); d++ ) {
84  curBC.push_back( resultShape->dim( d ));
85  }
86  curBC.push_back( resultCell );
87 
88  std::vector< vm_word > op_bytecode = logits->getOp()->getOpBytecode(
89  resCells, resultCell
90  );
91  curBC.insert(
92  std::end( curBC ), std::begin( op_bytecode ),
93  std::end( op_bytecode ));
94 
95  for ( int i = 0; i < logits->getIncomingNodes().size(); i++ ) {
96  auto derivShape = logits->getOp()->getDerivativeShape( i, predShapes );
97  auto derivTensor = new Tensor( derivShape, DataType::FLOAT ); // todo type
98  vm_word derivCell = virtualMemory->allocate( derivTensor );
99 
100  curBC.push_back( static_cast<vm_word>( OpCode::ALLOC ));
101  curBC.push_back( derivShape.dimensions());
102  for ( unsigned int d = 0; d < derivShape.dimensions(); d++ ) {
103  curBC.push_back( derivShape.dim( d ));
104  }
105  curBC.push_back( derivCell );
106 
107  std::vector< vm_word > deriv_bytecode =
108  logits->getOp()->getDerivativeBytecode(
109  i, resCells, derivCell
110  );
111  logits->addDerivative( derivCell );
112  curBC.insert(
113  std::end( curBC ),
114  std::begin( deriv_bytecode ),
115  std::end( deriv_bytecode ));
116  }
117 
118  logits->setCalculated( resultCell );
119 
120  }
121  }
122 
123  return std::make_tuple( curBC, resultShape, resultCell );
124 }
