25 
26  bool allocated = false;
27 
28  while ( cur != nullptr && !allocated ) {
29  if ( !cur->isUsed && (cur->endAddress - cur->startAddress) >= memNeeded ) {
30  auto newUsedBlock = new VMemoryBlock;
31  newUsedBlock->isUsed = true;
32  newUsedBlock->startAddress = cur->startAddress;
33  newUsedBlock->endAddress = newUsedBlock->startAddress + memNeeded;
34  newUsedBlock->prevBlock = cur->prevBlock;
35 
36  if ( cur->prevBlock != nullptr ) {
37  cur->prevBlock->nextBlock = newUsedBlock;
38  }
39 
40  if ( cur->endAddress - cur->startAddress > memNeeded ) {
41  auto freeBlock = new VMemoryBlock;
42  freeBlock->isUsed = false;
43  freeBlock->startAddress = newUsedBlock->endAddress;
44  freeBlock->endAddress = cur->endAddress;
45  freeBlock->nextBlock = cur->nextBlock;
46  freeBlock->prevBlock = newUsedBlock;
47 
48  newUsedBlock->nextBlock = freeBlock;
49 
50  if ( cur->nextBlock != nullptr ) {
51  cur->nextBlock->prevBlock = freeBlock;
52  }
53  } else {
54  newUsedBlock->nextBlock = cur->nextBlock;
55  if ( cur->nextBlock != nullptr ) {
56  cur->nextBlock->prevBlock = newUsedBlock;
57  }
58  }
59 
60  allocated = true;
61  curMemoryUsage += memNeeded;
62  if ( curMemoryUsage > maxMemoryUsage ) {
63  maxMemoryUsage = curMemoryUsage;
64  }
65

67  cur->prevBlock = nullptr;

69  head = newUsedBlock;
70  }
71  delete cur;
72  cur = newUsedBlock;
73  } else {
74  cur = cur->nextBlock;
75  }
76  }
77

80  }
